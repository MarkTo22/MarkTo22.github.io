### 设计模式

#### 1.单例模式

​			*只允许实例化一次的对象类【常用于：命名空间、资源优化、特定业务场景】*

​	单利模式（命名空间、静态变量）：

```js
var Single = (function(){
    // 私有变量
    var priv = {
        COUNT: 0,
        NAME: 'single-1',
        VERSION: '1.0.0'
    }
    
    return {
        get: function(name) {
            return priv[name] === undefined ? null : priv[name]
        },
        set: function(name, value) {
            priv[name] = value;
        }
    }
    
})()

console.log(Single.get('COUNT')) // 0
```

​	惰性单利：

```js
var LazySingle = (function(){
    var _instance = null;
    
    // 单例实例方法
    function Single(){
        return {
            name: 'single-2',
            version: '1.0.0',
            someMethod: function(){
                //...
            }
        }
    }
    return function() {
        if(!_instance) {
            _instance = Single();
        }
        return _instance;
    }
})()
```

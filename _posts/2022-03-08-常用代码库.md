1. **防抖**

   ```js
   function debounce(fn, delay) {
       var timer = null;
       return function() {
           var context = this;
           var args = arguments;
           if( timer ) clearTimeout(timer);
           timer = setTimeout(function(){
               fn.apply(context, args);
           }, delay);
       }
   }
   ```

2. **节流**

   ```js
   // 时间戳版本【执行时机：delay内，开始执行】
   function throttle(fn, delay) {
       var pre = 0;
       return function() {
           var context = this;
           var args = arguments;
           var now = Date.now();
           if( now - pre > delay ) {
               fn.apply(context, args);
               pre = now;
   		}
       }
   }
   // 定时器版本【执行时机：delay内，结束执行】
   function throttle2(fn, delay) {
       var timeout;
       return function() {
           var context = this;
           var args = arguments;
           if( !timeout ) {
               timeout = setTimeout(function(){
                   fn.apply(context, args);
                   timeout = null;
               }, delay);
           }
       }
   }
   ```

   // JSON.parse【不能处理：循环引用、undefined、function、symbol、Date、RegExp、NaN...】
   function cloneDeep(obj) {
       return JSON.parse(JSON.stringify(obj));
   }
   // 递归深拷贝【推荐使用：lodash.cloneDeep】
   function cloneDeep2(target, map = new WeakMap()) {
       var cloneObj = new target.constructor();
       if( target === null ) return target;
       if( target instanceof Date ) return new Date(target);
       if( target instanceof RegExp ) return new RegExp(target);
       if( typeof target !== 'object' ) return target;
       if( map.get(target) ){
           return map.get(target);
   	}
       map.set(target, cloneObj);
       for(var k in target) {
           if( target.hasOwnProperty(k) ) {
               cloneObj[k] = cloneDeep2(target[k], map)
           }
       }
       return cloneObj;
   }

4. **cookie工具（JS高程Ⅲ）**

   ```js
   var cookieUtils = {
   	get: function(name) {
   		var cookieStr = document.cookie;
   		var cookieName = encodeURIComponent(name) + "=",
   			cookieStart = cookieStr.indexOf(cookieName),
   			cookieValue = null;
   		if( cookieStart > -1 ) {
   			var cookieEnd = cookieStr.indexOf(";", cookieStart);
   			if( cookieEnd === -1 ) {
   				cookieEnd = cookieStr.length;
   			}
   			cookieValue = decodeURIComponent( cookieStr.substring(cookieStart+cookieName.length, cookieEnd) );
   		}
   		return cookieValue;
   	},
   	set: function(opt) {
   		if( !opt ) {
   			return;
   		}
   		var name = opt.name,
   			value = opt.value,
   			expires = opt.expires,
   			path = opt.path,
   			domain = opt.domain,
   			secure = opt.secure;
   		var cookieText = encodeURIComponent(name) + "=" + encodeURIComponent(value);
   		if( expires instanceof Date ) {
   			cookieText += "; expires=" + expires.toGMTString();
   		}
   		if( path ) {
   			cookieText += "; path=" + path;
   		}
   		if( domain ) {
   			cookieText += "; domain=" + domain;
   		}
   		if( secure ) {
   			cookieText += "; secure";
   		}
   		document.cookie = cookieText;
   	},
   	unset: function(opt) {
   		if( !opt ) {
   			return;
   		}
   		var name = opt.name,
   			path = opt.path,
   			domain = opt.domain,
   			secure = opt.secure;
   		this.set({
   			name: name,
   			value: "",
   			path: path,
   			domain: domain,
   			secure: secure
   		})
   	}
   }
   ```
   
5. **函数柯里化**

   ```js
   function curry(fn, ...args){
       const self = this;
       const len = fn.length;
       return function(...reset){
           let params = [...args, ...reset];
           if( params.length < len ) {
               return curry.call(self, fn, ...params);
           }else {
               return fn.apply(this, params);
           }
       }
   }
   // test
   function sum(a, b) {
       return a + b;
   }
   let _sum = curry(sum);
   _sum(1,2); // 3
   _sum(1)(2); // 3
   ```

   function flatten(arr, res = []) {
       for(let item of arr) {
           if( Array.isArray(item) ) {
               flatten(item, res);

   const getBornInfoByIdcard = (idcard) => {
   	let bornInfo = {
   		birth: '',
   		age: '',
   		sex: ''
   	}
   	let reg_idcard = /(^\d{15}$)|(^\d{18}$)|(^\d{17}(\d|X|x)$)/
   	idcard = idcard.toString
   	if( !reg_idcard.test(idcard) ){
   		return bornInfo
   	}
   	const len = idcard.length
   	const year = len === 18 ? idcard.substring(6,10) : ("19"+idcard.substring(6,8))
   	const month = len === 18 ? idcard.substring(10,12) : idcard.substring(8,10)
   	const day = len === 18 ? idcard.substring(12,14) : idcard.substring(10,12)
   	const now = new Date()
   	const nowMonth = now.getMonth() + 1
   	const nowDay = now.getDate()
   
   	const birth = year + '-' + month + '-' + day
   	const sex = len === 18 ? (idcard.substring(16,17) % 2 === 1 ? "男":"女"):(idcard.substring(14,15) % 2 === 1 ? "男":"女" )
   	let age = now.getFullYear() - year
   	if( nowMonth < month || (nowMonth == month && nowDay < day) ){
   		age--
   	}
   	return {
   		birth,
   		age,
   		sex
   	}
   }

2. **手写 bind/call/apply**

   ```js
   // bind
   Function.prototype._bind = function(ctx, ...args) {
       const self = this;
       const fn = function(...rest) {
           return self.call(ctx, ...args, ...rest)
       }
       if( self.prototype ) {
           fn.prototype = Object.create(self.prototype);
       }
       return fn;
   }
   // call
   Function.prototype._call = function(ctx, ...args) {
       const self = ctx == undefined ? window : Object(ctx);
       const key = Symbol();
       self[key] = this;
       const res = self[key](...args);
       delete self[key];
       return res;
   }
   // apply
   Function.prototype._apply = function(ctx, arr=[]) {
       const self = ctx == undefined ? window : Object(ctx);
       const key = Symbol();
       self[key] = this;
       const res = self[key](...arr);
       delete self[key];
       return res;
   }
   ```
